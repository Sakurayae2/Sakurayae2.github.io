# 1.添加

# Array.unshift() / Array.push()

  • 向数组首位 / 末尾添加一个或多个元素;
  • 返回新的数组长度;
  • 改变原数组;
let arr1 = [1, 2, 3];
arr1.unshift(0);
console.log(arr1); // [0,1,2,3]

let arr2 = [1, 2, 3];
arr2.push(4);
console.log(arr2); // [1,2,3,4]

# 2.删除

# Array.shift() / Array.pop()

  • 删除数组首位 / 末尾一个元素;
  • 返回首位 / 末尾的元素值，若数组为空，返回 undefined;
  • 改变原数组;
let arr1 = [1, 2, 3, 4];
arr1.shift();
console.log(arr1); // [2,3,4]

let arr2 = [1, 2, 3, 4];
arr2.pop();
console.log(arr2); // [1,2,3]

# Array.splice(start[])

  • 删除或替换现有元素或者原地添加新的元素来修改数组
  • 以数组形式返回被修改的内容
  • 改变原数组

# 3.排序

# Array.sort()

  • 默认按字符串 Unicode 码排序
  • 可以指定按某种顺序进行排列的函数
  • 改变原数组

  • # 默认排序

let arr1 = [1000, 3, 20, 5, 400];
arr1.sort();
console.log(arr1); // [1000,20,3,400,5]

  • # 从小到大

let arr2 = [1000, 3, 20, 5, 400];
arr2.sort(function (a, b) {
  return a - b;
});
arr2.sort((a, b) => a - b);
console.log(arr2); // [3,5,20,400,1000]

  • # 从大到小

let arr3 = [1000, 3, 20, 5, 400];
arr3.sort(function (a, b) {
  return b - a;
});
arr3.sort((a, b) => b - a);
console.log(arr3); // [1000,400,20,5,3]

  • # 对象可以按照某个属性进行排序

// sort by id
let arr = [
  { id: 2, name: "a" },
  { id: 1, name: "c" },
  { id: 3, name: "b" },
];
arr.sort((a, b) => a.id - b.id);
console.log(arr); // [{id:1,name:c},{id:2,name:'a'},{id:3,name:'b'}]

// sort by name
arr.sort(function (a, b) {
  var nameA = a.name.toUpperCase();
  var nameB = b.name.toUpperCase();
  if (nameA < nameB) {
    return -1;
  } else if (nameA > nameB) {
    return 1;
  }
  return 0;
});
console.log(arr); // [{id:2,name:'a'},{id:3,name:'b'},{id:1,name:'c'}]

# 4.合并

# Array.concat(arr1,arr2...)

  • 合并两个或多个数组;
  • 返回新的数组;
  • 不改变原数组;
let arr1 = [1, 2],
  arr2 = ["a", "b"],
  arr3 = [],
  arr = arr1.concat(arr2, arr3);
console.log(arr); // [1,2,'a','b']

# 5.翻转

# Array.reverse()

  • 翻转原数组;
  • 改变原数组;
let arr = [1, 2, 3, 4, 5];
arr.reverse();
console.log(arr); // [5,4,3,2,1]
